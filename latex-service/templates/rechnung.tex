% ─── Positionen (4 Spalten — RAW-Zeilen erwarten dieses Format) ──────────────
% Angelehnt ans Solar-Union-Vorbild: kein Grau-/Farbbalken mehr im Kopf, nur
% schlichte fette Beschriftung + Linie (booktabs). Die Einheiten stehen einmal
% im Spaltenkopf ("Menge (kWh)", "Tarif (ct/kWh)") statt in jeder Zeile neu --
% dafür ist "Position" jetzt die mit Abstand breiteste Spalte. Zusatzzeilen
% ohne Energiemenge (Mitgliedsbeitrag, Rabatte) lassen Menge/Tarif bewusst
% leer, genau wie die "Mitgliedschaft"-Zeile im Vorbild.
%
% Mehrere Zählpunkte pro Mitglied (z.B. zwei PV-Anlagen oder zwei Bezugs-
% zählpunkte): RAW_BEZUG_POSITIONEN_LISTE / RAW_EINSPEISUNG_POSITIONEN_LISTE
% nehmen dafür jeweils EINE vorformatierte Zeile PRO Zählpunkt auf (siehe
% php-erweiterung-rechnungsvariablen.md) -- jede Zeile zeigt den Produktnamen
% und darunter, als zweite (kleinere, graue) Zeile IN DERSELBEN Zelle, die
% 33-stellige Zählpunktnummer; Menge/Tarif/Betrag gehören zu dieser einen
% Zählpunkt-Zeile. Ohne PHP-Erweiterung (RAW-Liste leer) zeigt die Vorlage
% automatisch die alte Einzeiler-Fallback-Zeile (optional mit Zählpunktnummer
% über BEZUG_ZAEHLPUNKT/EINSPEISUNG_ZAEHLPUNKT).
% Spaltenbreiten: 82+20+24+26mm Inhalt + 3×6mm Abstand = 170mm = \textwidth.
{\renewcommand{\arraystretch}{1.3}%
\begin{tabular}{@{}>{\raggedright\arraybackslash}p{82mm}@{\hspace{6mm}}%
                    >{\raggedleft\arraybackslash}p{20mm}@{\hspace{6mm}}%
                    >{\raggedleft\arraybackslash}p{24mm}@{\hspace{6mm}}%
                    >{\raggedleft\arraybackslash}p{26mm}@{}}
  \toprule
  {\footnotesize\bfseries\color{navy}Position} &
  {\footnotesize\bfseries\color{navy}Menge (kWh)} &
  {\footnotesize\bfseries\color{navy}Tarif (ct/kWh)} &
  {\footnotesize\bfseries\color{navy}Betrag (EUR)}\\
  \midrule
  % WICHTIG: \newline statt \\ für den Umbruch INNERHALB der Zelle zur
  % Zählpunktnummer -- \\ würde (auch in einer p-Spalte!) die ganze
  % Tabellenzeile beenden statt nur innerhalb der Zelle umzubrechen, dadurch
  % würden Menge/Tarif/Betrag in die NÄCHSTE Zeile hineinrutschen.
  \ifx\rawbezugliste\empty
    Energiebezug aus der Gemeinschaft%
    \ifx\bezugzp\empty\else\newline{\footnotesize\color{midgray}Zählpunkt: \bezugzp}\fi
    & <<<BEZUG_KWH>>> & <<<BEZUG_TARIF>>> & <<<BEZUG_BETRAG>>>\\
  \else
    \rawbezugliste
  \fi
  \arrayrulecolor{rulegray}\hline
  \ifx\raweinspeisungliste\empty
    Einspeisevergütung (Gutschrift)%
    \ifx\einspeisungzp\empty\else\newline{\footnotesize\color{midgray}Zählpunkt: \einspeisungzp}\fi
    & <<<EINSPEISUNG_KWH>>> & <<<EINSPEISUNG_TARIF>>> & $-$\,<<<EINSPEISUNG_BETRAG>>>\\
  \else
    \raweinspeisungliste
  \fi
  \hline
  Mitgliedsbeitrag & & & <<<MITGLIEDSBEITRAG>>>\\
  <<<RAW_ZUSATZPOSITIONEN_LISTE>>>%
